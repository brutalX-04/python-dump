import requests, json

def fetch(url, cookies):
	data = {
		"url": url,
		"cookies": cookies
	}

	response = requests.post("http://brutalx.my.id/api/instagram/", json=data).json()
	return response

url = "https://www.instagram.com/p/C7_l9qUywlC/?__a=1&__d=dis"
cookies = 'your_cookie_instagram'
data = fetch(url, cookies)
print(data)
